    # Clipboard 剪贴板

    剪贴板操作方法。

    # 基础用法

    Clipboard 使用全局指令方式调用，$clipboard方法接受一个参数（需要赋值的文本），返回一个 Promise 对象

    <template>
  <div>
    <div class="input-wrap">
      <meg-input
        type="textarea"
        :rows="2"
        placeholder="请输入内容"
        v-model="text">
      </meg-input>
    </div>
    <div>
      <meg-button
        @click="onClickCopy"
        type="primary">
        copy
      </meg-button>
    </div>
  </div>
</template>
<script>
export default {
  data() {
    return {
      text: '这是要复制的文本'
    };
  },
  methods: {
    onClickCopy() {
      this.$clipboard(this.text).then(() => {
        this.$message({
          type: 'success',
          message: 'copy finished'
        });
      }, () => {
        this.$message({
          type: 'error',
          message: 'copy fail'
        });
      });
    }
  }
};
</script>
<style lang="scss" scoped>
  .input-wrap{
    width: 40%;
    margin: 10px 0 10px 0;
  }
</style>
