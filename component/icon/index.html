    # Icon 图标集

    基于vue-svgicon封装的通用的Vue Icon组件。

    # 基础用法

    详细使用文档参考：megvii-icons;


    请不要随意上传跟项目无关的图标

    • 文件名最好遵循 项目简称-文件名-类型，类型包括 stroke 和 fill，方便识别和分类
    • gitlab 仓库今后充当 icons 的管理后端，图标的修改、删除、重命名等操作不在 web 端暴露，避免对使用中的图标的修改，造成不可预知的问题，如果需要相关操作，联系管理员
    • 如果需要替换或删除图标，请保证这个图标没有被发版后的项目使用，在不确定的情况下修改图标，可能导致有些项目里的图标连带着被修改了，所以如果要修改图标，只能修改很明确不会对其他项目造成影响的；比如正在开发中的项目，之前上传的图标有变化需要更新图标，这时候可以删除或替换
    将svg文件拖到此处，或点击上传

    图标集合

    <template>
  <div class="container">
    <meg-upload
      class="upload-demo"
      drag
      multiple
      action=""
      accept="image/svg+xml"
      :http-request="submit"
      :show-file-list="false"
      :before-upload="beforeUpload">
      <meg-icon
        name="meg-icon-upload"
        class="meg-icon-upload"></meg-icon>
      <div class="meg-upload__text">
        将svg文件拖到此处，或<em>点击上传</em>
      </div>
    </meg-upload>
    <div>
      <h1>图标集合</h1>
    </div>
    <meg-collapse
      v-model="active"
      :sticky="true">
      <template
        v-for="(item, ins) in iconList">
        <meg-collapse-item
          v-if="item.length"
          :key="ins"
          :title="ins == 0 ? '其他' : ins"
          :name="ins">
          <ul class="icon-list">
            <li
              v-for="(icon, i) of item"
              :key="i"
              @click="copy(icon.name)">
              <div class="icon">
                <meg-icon
                  :svg-data="resetViewbox(icon)"
                  width="40px"
                  height="40px"></meg-icon>
              </div>
              <div class="icon-name">
                {{ icon.name }}
              </div>
            </li>
          </ul>
        </meg-collapse-item>
      </template>
    </meg-collapse>
  </div>
</template>
<script>
export default {
  data() {
    return {
      active: [],
      iconList: {}
    };
  },
  async mounted() {
    try {
      const res = await fetch('http://10.199.1.17:3100/icons/svg-icons');
      const { success, result } = await res.json();
      if (success) {
        this.iconList = result;
        this.active = Object.keys(result);
      } else {
        this.$message.error(result.message);
      }
    } catch (e) {
      console.log(e);
      this.$message.error(e.message);
    }
  },
  methods: {
    async copy(text) {
      const iconText = `<meg-icon name="${text}"></meg-icon>`;
      await this.$clipboard(iconText);
      const h = this.$createElement;
      this.$message({
        dangerouslyUseHTMLString: true,
        showClose: true,
        message: h('div', null, [
          h('span', {
            class: 'high-text'
          }, [
            iconText
          ]),
          ' copied.'
        ]),
        type: 'success'
      });
    },
    async submit({ file }) {
      const fd = new FormData();
      fd.append('file', file);
      try {
        const res = await fetch('http://10.199.1.17:3100/icons/upload', {
          method: 'POST',
          body: fd
        });
        const { success, result } = await res.json();
        if (success) {
          const keys = 'abcdefghijklmnopqrstuvwxyz0';
          const { name } = result;
          const prefix = name.slice(0, 1);
          const key = keys.includes(prefix) ? prefix : 0;
          this.iconList[key].unshift(result);
          if (!this.active.includes(key)) {
            this.active.push(key);
          }
          this.$message.success('上传成功');
        } else {
          this.$message.error(result.message);
        }
      } catch (e) {
        console.log(e);
        this.$message.error(e.message);
      }
    },
    beforeUpload(file) {
      const valid = file.type.includes('image/svg');
      if (!valid) {
        this.$message.warning('请选择svg文件上传');
      }
      return valid;
    },
    resetViewbox(v) {
      const vb = Object.assign({}, v);
      const val = v.viewBox;
      vb.viewBox = val.slice(1, val.length - 1);
      return vb;
    }
  }
};
</script>
